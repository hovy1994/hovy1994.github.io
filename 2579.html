#include
#include
using namespace std;

int dp[3001];
int stair[3001];
int main()
{
int n;

scanf("%d", &n);
for (int i = 0; i < n; i++)
{
scanf("%d", &stair[i]);
}

dp[0] = stair[0];
dp[1] = dp[0]+stair[1];
dp[2] = stair[2] + max(stair[0], stair[1]);

for (int i = 3; i < n; i++)
{
dp[i] = max(stair[i] + stair[i - 1] + dp[i - 3], stair[i] + dp[i - 2]);
}

printf("%d\n", dp[n - 1]);

}
