{

{

for (int j = 0; j < sentence.size(); j++)
{
if (j+1 continue;
else if (sentence[j] < 58)
{
while (j {
temp += sentence[j];

}
v.push_back(make_pair(temp.size(), temp));

}
}
}

sort(v.begin(), v.end());

for (int i = 0; i < v.size(); i++)
{
cout << v[i].second << "\n";
}

}
